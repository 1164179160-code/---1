  var st = useState('ha');
  var searchText = st[0], setSearchText = st[1];

  var renderItem = function(info) {
    var item = info.item;
    return (
      <TouchableOpacity style={styles.listItem} activeOpacity={0.6}>
        <View style={styles.listItemLeft}>
          <View style={styles.locationIcon}>
            <View style={styles.locationPin} />
            <View style={styles.locationDot} />

          <Text style={styles.locationName} numberOfLines={1}>{item.name}</Text>
          <Text style={styles.locationDistance}>{item.distance}</Text>
        </View>
        <Text style={styles.chevron}>›</Text>

  };

  return (
    <SafeAreaView style={styles.container}>
      <StatusBar barStyle="dark-content" />

        <TouchableOpacity style={styles.backButton} activeOpacity={0.7}>

        </TouchableOpacity>
        <View style={styles.searchBar}>
          <Text style={styles.searchIcon}>🔍</Text>
          <TextInput
            style={styles.searchInput}
            value={searchText}
            onChangeText={setSearchText}
            placeholder="搜索地点"
            placeholderTextColor="#BCC4CC"
            autoFocus
          />
          {searchText.length > 0 && (
            <TouchableOpacity onPress={function() { setSearchText(''); }} style={styles.clearButton}>
              <View style={styles.clearIcon}>
                <Text style={styles.clearIconText}>×</Text>
              </View>
            </TouchableOpacity>
          )}
          <View style={styles.searchDivider} />
          <TouchableOpacity style={styles.searchButton}>
            <Text style={styles.searchButtonText}>搜索</Text>
          </TouchableOpacity>
        </View>
        <View style={styles.capsuleButton}>
          <View style={styles.capsuleDots}>
            <View style={styles.capsuleDotSmall} />
            <View style={styles.capsuleDotLarge} />
          </View>
          <View style={styles.capsuleDivider} />
          <View style={styles.capsuleCircle}>
            <View style={styles.capsuleCircleInner} />
          </View>
        </View>

        renderItem={renderItem}
        keyExtractor={function(item) { return item.id; }}
        style={styles.list}
        ItemSeparatorComponent={function() { return <View style={styles.separator} />; }}

    </SafeAreaView>

}

交互与动画

交互设计细节

搜索框聚焦：自动弹出键盘，输入即搜索（无需点击搜索按钮）
搜索建议：实时匹配，高亮关键词
历史记录：最近 5 条，左滑删除单条，一键清空
网点卡片：距离 + 可用车位 + 营业状态，信息层级清晰

项目概述

还车网点搜索页是蓝莓骑行产品体验闭环的关键一环。用户骑行结束后，需要快速找到最近的还车网点。这个页面要做到搜索快、信息清晰、操作简单。

交互设计

  • 搜索框：自动聚焦、实时搜索建议、清除按钮
  • 搜索历史：最近搜索记录，一键清除
  • 推荐网点：基于位置的智能推荐
  • 网点卡片：名称、距离、可用车位、营业时间
  • 地图联动：搜索结果与地图标注同步

设计思考

还车场景的用户心理是"快速完成任务"，所以整个页面的设计原则是"减少思考、减少操作"。搜索框默认展开、历史记录置顶、推荐网点按距离排序——每一个决策都是为了让用户最快找到目标。

在产品体验中的位置

如果说极速骑行是"让人想骑"，蓝莓骑行首页是"让人选车"，那还车网点就是"让人安心还"。一个完整的骑行产品体验，必须覆盖从"想骑"到"还车"的全链路。这个页面补齐了最后一环。
